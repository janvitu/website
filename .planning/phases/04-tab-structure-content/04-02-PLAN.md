**File 1: Create `hugo/assets/ts/tabs.ts`**

New file with a single exported function `initTabs`:

```typescript
export const initTabs = (): void => {
	const tabButtons = document.querySelectorAll<HTMLButtonElement>("[role='tab']");
	const tabPanels = document.querySelectorAll<HTMLDivElement>("[role='tabpanel']");

	if (tabButtons.length === 0) return;

	// Defensive: ensure first panel is visible on load (matches shortcode active="true")
	const firstButton = tabButtons[0];
	const firstPanel = tabPanels[0];
	if (firstButton && firstPanel) {
		firstButton.setAttribute("aria-selected", "true");
		firstButton.style.color = "var(--color-text)";
		firstButton.style.borderBottomColor = "var(--color-accent)";
		firstPanel.hidden = false;
	}

	tabButtons.forEach((button) => {
		button.addEventListener("click", () => {
			const targetId = button.getAttribute("data-tab");
			if (!targetId) return;

			// Deactivate all tabs
			tabButtons.forEach((btn) => {
				btn.setAttribute("aria-selected", "false");
				btn.style.color = "var(--color-text-muted)";
				btn.style.borderBottomColor = "transparent";
			});

			// Hide all panels
			tabPanels.forEach((panel) => {
				panel.hidden = true;
			});

			// Activate clicked tab
			button.setAttribute("aria-selected", "true");
			button.style.color = "var(--color-text)";
			button.style.borderBottomColor = "var(--color-accent)";

			// Show target panel
			const targetPanel = document.getElementById(`panel-${targetId}`);
			if (targetPanel) {
				targetPanel.hidden = false;
			}
		});
	});
};
```

Key details:
- Named export `initTabs` (not default export -- per project convention)
- Uses `querySelectorAll` with ARIA role selectors (matches DOM from Plan 01)
- Style changes via `button.style.color` and `button.style.borderBottomColor` (inline styles, consistent with terminal chrome pattern per UI-SPEC Implementation Note 4)
- Active state: `color: var(--color-text)`, `borderBottomColor: var(--color-accent)` (per D-03, UI-SPEC Color table)
- Inactive state: `color: var(--color-text-muted)`, `borderBottomColor: transparent` (per D-03, UI-SPEC Tab Button table)
- Toggles `hidden` attribute on panels (per D-06, RESEARCH.md Pattern 1)
- Updates `aria-selected` for accessibility
- Early return if no tab buttons found (graceful on non-tab pages)
- Defensive first-panel activation on load (belt-and-suspenders with shortcode `active="true"`)
- No `console.log` calls (per project convention)
- Tab indentation, double quotes, trailing commas (per Prettier config)
- No URL hash sync, no transitions (those are Phase 5 per D-07)
